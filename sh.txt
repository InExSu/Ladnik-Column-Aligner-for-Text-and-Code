# Установить права на файлы проекта на текущего пользователя
# И дать файлам sh разрешение на запусе
chown -R "$(whoami):$(id -gn)" .
find . -name "*.sh" -exec chmod +x {} +

# Analyze with current configuration
composer analyse

# Analyze with different level (0-9, where 9 is strictest)
php vendor/bin/phpstan analyse --level 7

# Analyze specific files
php vendor/bin/phpstan analyse a_Do.php a_Methods.php

# Run all tests
php vendor/bin/phpunit tests/

# Run specific test file
php vendor/bin/phpunit tests/CompanyLinkerUnitTest.php
